                 LINK_LIBRARIES Boost RELAX DetDescLib DAQEventLib RecEvent GaudiAlgLib HltInterfaces LHCbKernel)

                     dict/LumiAlgsDict.h
                     dict/LumiAlgsDict.xml
                     LINK_LIBRARIES Boost GaudiKernel
                     OPTIONS "-U__MINGW32__")
